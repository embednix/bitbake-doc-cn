# Define SRC_URI with two different Git repositories
SRC_URI = " \
    git://git.example.com/repo1.git;branch=main;name=repo1 \
    git://git.example.com/repo2.git;branch=develop;name=repo2 \
"

# Define SRCREV for each repository
SRCREV_repo1 = "${AUTOINC}+abcdef1234567890"
SRCREV_repo2 = "${AUTOINC}+1234567890abcdef"

# Define SRCREV_FORMAT to concatenate the revisions
SRCREV_FORMAT = "repo1_repo2"
